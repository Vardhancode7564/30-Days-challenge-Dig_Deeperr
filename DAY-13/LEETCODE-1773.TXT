class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue){
        int har=0;
        if(ruleKey=="color") har=1;
        else if(ruleKey=="name") har=2;

        int ans=0;
        for(auto & item:items){
            if(item[har]==ruleValue){
                ans++;
            }
        }
        return ans;
    }
};
