class Solution {
public:
    int lengthOfLastWord(string s) {
        int n=s.length(),len=0;
        bool var=false;

        for(int i=n-1;i>=0;i--){
            if(s[i]!=' '){
                len++;
                var=true;
            }
            else if(var){
                break;
            }
        }
        return len;
    }
};
