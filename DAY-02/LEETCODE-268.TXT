class Solution {
public:
    int missingNumber(vector<int>& nums) {
        // int xor1=0;
        // int xor2=0;
        // int n=nums.size();
        // for(int i=0;i<n;i++){
        //     xor2^=nums[i];
        //     xor1^=(i+1);
        // }
        
        // return xor1^xor2;
        int n=nums.size();
        int sum=0;
        int sum1=n*(n+1)/2;
        for(int i=0;i<n;i++){
            sum+=nums[i];
        }
        return sum1-sum;
    }
};
