class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        if (strs.empty()) return "";
        
        string har=strs[0];
        
        for (int i=1;i<strs.size();i++) {
            while(strs[i].find(har)!=0) { 
                har.pop_back();              
                if(har.empty()) return "";
            }
        }
        return har;
    }
};
